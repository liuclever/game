def king_register():
    """报名参加本周挑战赛"""
    user_id = get_current_user_id()

    today = datetime.now()

    rank_info = execute_query(
        "SELECT * FROM king_challenge_rank WHERE user_id = %s",
        (user_id,)
    )
    
    if rank_info and rank_info[0]['is_registered']:
        return jsonify({"ok": False, "error": "本周已报名"})
    
    # 更新报名状态
    execute_update(
        "UPDATE king_challenge_rank SET is_registered = 1 WHERE user_id = %s",
        (user_id,)
    )
    
    return jsonify({"ok": True, "message": "报名成功！"})

def advance_to_finals():

#### 5.2 正赛匹配
```python
def run_final_stage(stage):
    """执行正赛某一轮次"""
    season = get_current_season()
    
    # 获取本轮参赛选手
    players = execute_query(
        """SELECT user_id FROM king_final_stage 
           WHERE season = %s AND stage = %s AND is_winner IS NULL""",
        (season, stage)
    )
    
    # 随机打乱
    import random
    random.shuffle(players)
    
    # 配对
    matches = []
    for i in range(0, len(players), 2):
        if i + 1 < len(players):
            # 正常配对
            matches.append((players[i], players[i+1]))
        else:
            # 轮空
            matches.append((players[i], None))
    
    # 执行战斗
    for p1, p2 in matches:
        if p2 is None:
            # 轮空，直接晋级
            winner = p1
        else:
            # 战斗
            result = do_pvp_battle(p1, p2)
            winner = result.winner_id
        
        # 记录结果
        next_stage = get_next_stage(stage)
        execute_update(
            """INSERT INTO king_final_stage 
               (season, user_id, stage) VALUES (%s, %s, %s)""",
            (season, winner, next_stage)
        )
        
        # 发放奖励
        send_stage_reward(winner, stage)
```
